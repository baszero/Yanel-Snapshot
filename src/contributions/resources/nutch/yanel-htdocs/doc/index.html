Nutch resource

Overview

Name:
nutch
Namespace:
http://www.wyona.org/yanel/resource/1.0
Universal Name:
<{http://www.wyona.org/yanel/resource/1.0}nutch/>
Description:
Getting Search results from a Nutch crawl (tested nutch version 0.8.x)
Java Class:
org.wyona.yanel.impl.resources.NutchResource

Examples

Search results with some XSLT applied using the default configuration src/contributions/resources/nutch/conf/nutch-local.xml

    <yanel:resource-config xmlns:yanel="http://www.wyona.org/yanel/rti/1.0">
      <yanel:rti name="nutch" namespace="http://www.wyona.org/yanel/resource/1.0"/>

      <yanel:property name="global-xslt" value="/xslt/global.xsl"/>
    </yanel:resource-config>

Search results with some XSLT applied using a custom configuration

    <yanel:resource-config xmlns:yanel="http://www.wyona.org/yanel/rti/1.0">
      <yanel:rti name="nutch" namespace="http://www.wyona.org/yanel/resource/1.0"/>

      <yanel:property name="global-xslt" value="/xslt/global.xsl"/>
      <yanel:property name="nutch-config" value="file:/home/USER/nutch-config.xml"/>
    </yanel:resource-config>

Search results with some XSLT applied using a grouped custom configuration (accessing multiple indexes)

    <yanel:resource-config xmlns:yanel="http://www.wyona.org/yanel/rti/1.0">
      <yanel:rti name="resource-creator" namespace="http://www.wyona.org/yanel/resource/1.0"/>

      <yanel:custom-config>
        <nr:configs>
          <nr:config name="ml-usage" src="file:/home/USER/yanel-website/content/nutch-config/nutch-local-yanel-ml-usage.xml"/>
          <nr:config name="ml-development" src="file:/home/USER/yanel-website/content/nutch-config/nutch-local-yanel-ml-development.xml"/>
        </nr:configs>

        <nr:groups>
          <nr:group name="usage" label="Usage" i18n-key="mlusage">
            <nr:nutch-config name="ml-usage"/>
          </nr:group>
          <nr:group name="dev" label="Development" i18n-key="mldevelopment" selected="true">
            <nr:nutch-config name="ml-development"/>
          </nr:group>
          <nr:group name="all" label="All Lists" i18n-key="all-lists">
            <nr:nutch-config name="ml-usage"/>
            <nr:nutch-config name="ml-development"/>
          </nr:group>
        </nr:groups>
      </yanel:custom-config>
    </yanel:resource-config>
