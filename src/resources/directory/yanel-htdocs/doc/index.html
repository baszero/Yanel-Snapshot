<{http://www.wyona.org/yanel/resource/1.0}directory/>

Examples

Overwrite the default XSLT

    <yanel:resource-config xmlns:yanel="http://www.wyona.org/yanel/rti/1.0">
      <yanel:rti name="directory" namespace="http://www.wyona.org/yanel/resource/1.0"/>

      <yanel:property name="default-xslt" value="/custom-default-xhtml.xsl"/>
      <yanel:property name="mime-type" value="application/xhtml+xml"/>

    </yanel:resource-config>

Embed default XHTML view into some global layout

    <yanel:resource-config xmlns:yanel="http://www.wyona.org/yanel/rti/1.0">
      <yanel:rti name="directory" namespace="http://www.wyona.org/yanel/resource/1.0"/>

      <yanel:property name="xslt" value="/global.xsl"/>
      <yanel:property name="mime-type" value="application/xhtml+xml"/>

    </yanel:resource-config>
